PROGRAM STATS
VAR
    SUM SUMSQ,I,VALUE,VALUE,MEAN,VARIANCE : INTEGER;
    A,B,C : REAL
BEGIN
    SU :=0;
    SUMSQ 0;
    FOR T := 1 TO 100 DO
        BEGIN
            READ(VALUE)
            SUM := SUM  VALUE;
            SUMSQ := SUMSQ + VALUE * B
